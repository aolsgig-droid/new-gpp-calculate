<script>


const lbsInput = document.getElementById("lbs");

const gppInput = document.getElementById("gpp");


function calculate(){


    const lbs = Number(lbsInput.value);

    const gpp = Number(gppInput.value);


    const arrow = Math.round(lbs * gpp * 0.0647989);


    let spine = "";


    if (lbs >= 30 && lbs <= 35){

        spine = 600;

    }

    else if (lbs >= 36 && lbs <= 39){

        spine = 550;

    }

    else if (lbs >= 40 && lbs <= 45){

        spine = 500;

    }

    else if (lbs >= 46 && lbs <= 49){

        spine = 450;

    }

    else if (lbs >= 50 && lbs <= 55){

        spine = 400;

    }

    else if (lbs >= 56 && lbs <= 59){

        spine = 350;

    }

    else if (lbs >= 60 && lbs <= 65){

        spine = 300;

    }

    else if (lbs >= 66 && lbs <= 69){

        spine = 250;

    }

    else if (lbs >= 70 && lbs <= 100){

        spine = 200;

    }


    document.getElementById("arrow").innerHTML =

        `Arrow: ${arrow.toLocaleString()} gram`;


    document.getElementById("spine").innerHTML =

        `Spine: ${spine}`;

}


lbsInput.addEventListener("input", calculate);

gppInput.addEventListener("input", calculate);


calculate();


</script>
